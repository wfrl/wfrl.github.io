Basic results

Definition. nat_eq
ℕ = ⋂{A | A ⊆ ℝ ∧ 0 ∈ A ∧ ∀n. n ∈ A → n + 1 ∈ A}
Theorem. nat_incl_in_real
ℕ ⊆ ℝ
Theorem. nat_is_set
set ℕ
Theorem. zero_in_nat
0 ∈ ℕ
Theorem. succ_in_nat
n ∈ ℕ → n + 1 ∈ ℕ
Theorem. nat_induction_sets
A ⊆ ℕ → 0 ∈ A → (∀n. n ∈ A → n + 1 ∈ A) → A = ℕ
Theorem. nat_induction
P 0 → (∀n. n ∈ ℕ → P n → P (n + 1)) → (∀n. n ∈ ℕ → P n)
Theorem. nat_non_negative
n ∈ ℕ → 0 ≤ n
Theorem. nat_succ_non_zero
n ∈ ℕ → ¬(n + 1 = 0)
Theorem. nat_succ_inj
a ∈ ℕ → b ∈ ℕ → a + 1 = b + 1 → a = b
Theorem. nat_rec_uniqueness
app f 0 = x0 ∧ (∀n. n ∈ ℕ → app f (n + 1) = φ n (app f n)) → app g 0 = x0 ∧ (∀n. n ∈ ℕ → app g (n + 1) = φ n (app g n)) → ∀n. n ∈ ℕ → app f n = app g n
Theorem. nat_rec_existence
set X → x0 ∈ X → (∀n. n ∈ ℕ → ∀x. x ∈ X → φ n x ∈ X) → ∃f. set f ∧ (map f ℕ X ∧ (app f 0 = x0 ∧ ∀n. n ∈ ℕ → app f (n + 1) = φ n (app f n)))
Theorem. nat_recursion
set X → x0 ∈ X → (∀n. n ∈ ℕ → ∀x. x ∈ X → φ n x ∈ X) → ∃g. set g ∧ ∀f. g = f ↔ map f ℕ X ∧ (app f 0 = x0 ∧ ∀n. n ∈ ℕ → app f (n + 1) = φ n (app f n))
Theorem. nadd_neutl
a ∈ ℕ → 0 + a = a
Theorem. nadd_neutr
a ∈ ℕ → a + 0 = a
Theorem. nsub_zero
a ∈ ℕ → a - 0 = a
Theorem. nmul_neutl
a ∈ ℕ → 1⋅a = a
Theorem. nmul_neutr
a ∈ ℕ → a⋅1 = a
Theorem. nmul_lzero
a ∈ ℕ → 0⋅a = 0
Theorem. nmul_rzero
a ∈ ℕ → a⋅0 = 0
Theorem. nadd_comm
a ∈ ℕ → b ∈ ℕ → a + b = b + a
Theorem. nmul_comm
a ∈ ℕ → b ∈ ℕ → a⋅b = b⋅a
Theorem. nadd_assoc
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → (a + b) + c = a + (b + c)
Theorem. nmul_distl_add
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a⋅(b + c) = a⋅b + a⋅c
Theorem. nmul_distr_add
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → (a + b)⋅c = a⋅c + b⋅c
Theorem. nmul_distl_sub
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a⋅(b - c) = a⋅b - a⋅c
Theorem. nmul_distr_sub
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → (a - b)⋅c = a⋅c - b⋅c
Theorem. nmul_compatl_neg
a ∈ ℕ → b ∈ ℕ → -(a⋅b) = (-a)⋅b
Theorem. nmul_compatr_neg
a ∈ ℕ → b ∈ ℕ → -(a⋅b) = a⋅(-b)
Theorem. nadd_cancel_rr
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a + c = b + c → a = b
Theorem. nadd_closed
a ∈ ℕ → b ∈ ℕ → a + b ∈ ℕ
Theorem. nmul_closed
a ∈ ℕ → b ∈ ℕ → a⋅b ∈ ℕ
Theorem. nat_zero_or_succ
n ∈ ℕ → n = 0 ∨ ∃m. m ∈ ℕ ∧ n = m + 1
Theorem. nat_non_zero_is_succ
n ∈ ℕ → ¬n = 0 → ∃m. m ∈ ℕ ∧ n = m + 1

Basic inequalities

Theorem. nle_refl
a ∈ ℕ → a ≤ a
Theorem. nle_antisym
a ∈ ℕ → b ∈ ℕ → a ≤ b → b ≤ a → a = b
Theorem. nle_trans
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a ≤ b → b ≤ c → a ≤ c
Theorem. nle_total
a ∈ ℕ → b ∈ ℕ → a ≤ b ∨ b ≤ a
Theorem. nle_compat_addl
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a ≤ b → c + a ≤ c + b
Theorem. nle_compat_add
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a ≤ b → a + c ≤ b + c
Theorem. nle_lt_trans
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a ≤ b → b < c → a < c
Theorem. nlt_lt_trans
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a < b → b < c → a < c
Theorem. nlt_add_cancel_rr
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a + c < b + c → a < b
Theorem. nle_lt_tricho
a ∈ ℕ → b ∈ ℕ → a ≤ b ∨ b < a
Theorem. nlt_succ
a ∈ ℕ → a < a + 1
Theorem. nlt_le_trans
a ∈ ℕ → b ∈ ℕ → c ∈ ℕ → a < b → b ≤ c → a < c
Theorem. nat_lt_from_succ_le
n ∈ ℕ → k ∈ ℕ → n + 1 ≤ k → n < k
Theorem. nat_succ_le_from_lt
n ∈ ℕ → k ∈ ℕ → n < k → n + 1 ≤ k
Theorem. nle_succ_split
k ∈ ℕ → n ∈ ℕ → k ≤ n + 1 → k ≤ n ∨ k = n + 1

Strong induction

Theorem. nat_strong_induction
P 0 → (∀n. n ∈ ℕ → (∀k. k ∈ ℕ → k ≤ n → P k) → P (n + 1)) → (∀n. n ∈ ℕ → P n)
Theorem. nat_strong_induction_terse
(∀n. n ∈ ℕ → (∀k. k ∈ ℕ → k < n → P k) → P n) → (∀n. n ∈ ℕ → P n)
