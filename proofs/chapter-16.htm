Basic results

Definition. int_eq
ℤ = {z | z ∈ ℝ ∧ ∃a. ∃b. a ∈ ℕ ∧ b ∈ ℕ ∧ z = a - b}
Theorem. int_incl_in_real
ℤ ⊆ ℝ
Theorem. nat_incl_in_int
ℕ ⊆ ℤ
Theorem. ineg_closed
a ∈ ℤ → -a ∈ ℤ
Theorem. iadd_closed
a ∈ ℤ → b ∈ ℤ → a + b ∈ ℤ
Theorem. isub_closed
a ∈ ℤ → b ∈ ℤ → a - b ∈ ℤ
Theorem. imul_closed
a ∈ ℤ → b ∈ ℤ → a⋅b ∈ ℤ
Theorem. iadd_neutl
a ∈ ℤ → 0 + a = a
Theorem. iadd_neutr
a ∈ ℤ → a + 0 = a
Theorem. isub_zero
a ∈ ℤ → a - 0 = a
Theorem. isub_self
a ∈ ℤ → a - a = 0
Theorem. imul_neutl
a ∈ ℤ → 1⋅a = a
Theorem. imul_neutr
a ∈ ℤ → a⋅1 = a
Theorem. imul_lzero
a ∈ ℤ → 0⋅a = 0
Theorem. imul_rzero
a ∈ ℤ → a⋅0 = 0
Theorem. iadd_comm
a ∈ ℤ → b ∈ ℤ → a + b = b + a
Theorem. imul_comm
a ∈ ℤ → b ∈ ℤ → a⋅b = b⋅a
Theorem. iadd_assoc
a ∈ ℤ → b ∈ ℤ → c ∈ ℤ → (a + b) + c = a + (b + c)
Theorem. imul_distl_add
a ∈ ℤ → b ∈ ℤ → c ∈ ℤ → a⋅(b + c) = a⋅b + a⋅c
Theorem. imul_distr_add
a ∈ ℤ → b ∈ ℤ → c ∈ ℤ → (a + b)⋅c = a⋅c + b⋅c
Theorem. imul_distl_sub
a ∈ ℤ → b ∈ ℤ → c ∈ ℤ → a⋅(b - c) = a⋅b - a⋅c
Theorem. imul_distr_sub
a ∈ ℤ → b ∈ ℤ → c ∈ ℤ → (a - b)⋅c = a⋅c - b⋅c
Theorem. imul_compatl_neg
a ∈ ℤ → b ∈ ℤ → -(a⋅b) = (-a)⋅b
Theorem. imul_compatr_neg
a ∈ ℤ → b ∈ ℤ → -(a⋅b) = a⋅(-b)
Theorem. isub_neg
a ∈ ℤ → b ∈ ℤ → -(a - b) = b - a
Theorem. isub_cancel_middle
a ∈ ℤ → b ∈ ℤ → c ∈ ℤ → (a - b) + (b - c) = a - c

a ∈ ℤ → b ∈ ℤ → a ≤ b → b ≤ a → a = b

Elementary number theory

Definition. divides_equi
divides m a ↔ ∃k. k ∈ ℤ ∧ a = m⋅k
Definition. cong_equi
cong a b m ↔ divides m (a - b)
Definition. divisors_eq
divisors a = {d | d ∈ ℕ ∧ divides d a}
Definition. prime_equi
prime a ↔ 2 ≤ a ∧ divisors a = {1, a}
Theorem. cong_intro
k ∈ ℤ → a - b = m⋅k → cong a b m
Theorem. cong_elim
cong a b m → ∃k. k ∈ ℤ ∧ (a - b) = m⋅k
Theorem. cong_refl
a ∈ ℤ → m ∈ ℤ → cong a a m
Theorem. cong_symm
a ∈ ℤ → b ∈ ℤ → m ∈ ℤ → cong a b m → cong b a m
Theorem. cong_trans
a ∈ ℤ → b ∈ ℤ → c ∈ ℤ → m ∈ ℤ → cong a b m → cong b c m → cong a c m
